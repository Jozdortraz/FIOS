## 提示I - One Pass

<details markdown="1">
<summary markdown="span">实现</summary>

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {number[]}
 */
const nextLargerNodes = function(head) {
  const stack = [];
  const result = [];

  let p = head;
  let counter = 0;
  while (p !== null) {
    const current = p.val;

    while (stack.length > 0 && stack[stack.length - 1].val < current) {
      const {index} = stack.pop();
      result[index] = current;
    }

    stack.push({
      index: counter,
      val: current
    });

    result.push(0);

    p = p.next;
    counter += 1;
  }

  return result;
};
```
